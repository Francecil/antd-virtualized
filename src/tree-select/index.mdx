### 综合实例：图标，默认展开

<Playground style={{height:"500px"}}>
<TreeSelect
  showSearch
  allowClear
  style={{ width: 300 }}
  placeholder="请输入"
  defaultExpandAll
  treeData={treeSelectData}
  onSelect={(value, node, extra) => console.log(value, node, extra)}
  keyField='key'
  titleField='title'
  selectable
  treeRender={(node)=> {
    if(node.children && node.children.length){
      return (<span>{node.title} ({node.children.length})</span>)
    } else {
      let icon = ""
      if(node.type==="1"){
        icon = "home"
      } else if(node.type==="2") {
        icon = "tag"
      } else if(node.type==="3") {
        icon = "edit"
      }
      return (<span><Icon type={icon}/> {node.title}</span>)
    }
  }}
/>
</Playground>
